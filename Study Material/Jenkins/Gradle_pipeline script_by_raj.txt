node {
    
    stage("Git Clone"){

        git credentialsId: 'GIT_HUB_CREDENTIALS', url: 'https://github.com/Divyaraj-Mandale/springboot-with-docker-kub.git'
    }
    
    stage('Gradle Build') {

       sh './gradlew build'
}
    stage("Docker build"){
        sh 'docker version'
        sh 'docker build -t jhooq-docker-demo .'
        sh 'docker image list'
        sh 'docker tag jhooq-docker-demo rajmandale/gradle:jhooq-docker-demo'
    }
    
    stage("Docker Login"){
        withCredentials([string(credentialsId: 'DOCKER_HUB_PASSWORD', variable: 'PASSWORD')]) {
            sh 'docker login -u rajmandale -p $PASSWORD'
        }
    } 
    
    stage("Push Image to Docker Hub"){
        sh 'docker push  rajmandale/gradle:jhooq-docker-demo'
    }
    
    stage("SSH Into k8s Server") {
        def remote = [:]
        remote.name = 'k8s-master'
        remote.host = '172.31.17.204'
        remote.user = 'root'
        remote.password = 'root'
        remote.allowAnyHosts = true
        
    stage('Put k8s-spring-boot-deployment.yml onto k8smaster') {
            sshPut remote: remote, from: 'k8s-spring-boot-deployment.yml', into: '.'
        } 
        
    stage('Deploy spring boot') {
          sshCommand remote: remote, command: "kubectl apply -f k8s-spring-boot-deployment.yml"
        }
        
    }

}
